//打印水仙花数
//
//作业内容
//求出0～100000之间的所有“水仙花数”并输出。
//
//“水仙花数”是指一个n位数，其各位数字的n次方之和确好等于该数本身，如 : 153＝1 ^ 3＋5 ^ 3＋3 ^ 3，则153是一个“水仙花数”。

#include <math.h>

int main()
{
	int i = 0;		
	for (i = 0; i < 100000; i++)
	{
		int n = 1;
		int tmp = i;
		//1.计算这个数字有n位数(1---10000)
		while (tmp/ 10 != 0)
		{
			tmp /= 10;
			n++;
		}
		int sum = 0;
		//2.这个数字的每一位数字的n次方求和
		tmp = i;
		while (tmp)
		{
		sum += pow(tmp % 10, n);
			tmp /=10;
		}

		//3.判断并筛选打印
		if (sum==i)
		printf("%d  ", i);
	}
	return 0;
}
